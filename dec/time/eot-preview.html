    def unix2date(unix):
    dote = unix / 86400 + 719468
    cykl = (
        dote if dote >= 0
        else dote - 146096
    ) // 146097
    dotc = dote - cykl * 146097
    yotc = (dotc
        - dotc // 1460
        + dotc // 36524
        - dotc // 146096
    ) // 365
    return [
        int(yotc + cykl * 400),
        int(dotc - (yotc * 365
            + yotc // 4
            - yotc // 100))
    ]

    def model(gamma, p0, p1, p2, p3, p4):
    return (
        p0 +
        p1 * np.cos(gamma) +
        p2 * np.sin(gamma) +
        p3 * np.cos(2 * gamma) +
        p4 * np.sin(2 * gamma)
    )
    In [8]:
